53    pub fn open<P: AsRef<Path>>(path: P) -> Result<Self> {
54        let path = path.as_ref().to_path_buf();
55        let mut file = BufReader::new(
56            File::open(&path)
57                .with_context(|| format!("Failed to open VPK file: {}", path.display()))?,
58        );
59
60        let header = Self::read_header(&mut file)?;
61        let tree = Self::read_file_tree(&mut file, &header)?;
62        let checksums = if header.version == VPKVersion::V2 {
63            Some(Self::read_checksums(&mut file, &header)?)
64        } else {
65            None
66        };
67
68        Ok(VPK {
69            path: Some(path),
70            header,
71            tree,
72            checksums,
73        })
74    }

76    /// Creates a new VPK from a directory structure
77    pub fn from_directory<P: AsRef<Path>>(path: P) -> Result<Self> {
78        let path = path.as_ref();
79        if !path.is_dir() {
80            bail!("Path is not a directory: {}", path.display());
81        }
82
83        let mut tree = HashMap::new();
84        // let mut file_count = 0;
85
86        // Walk the directory and build the file tree
87        for entry in WalkDir::new(path).into_iter().filter_map(|e| e.ok()) {
88            if entry.file_type().is_file() {
89                let relative_path = entry
90                    .path()
91                    .strip_prefix(path)
92                    .context("Failed to get relative path")?;
93
94                let path_str = normalize_path(&relative_path.to_string_lossy());
95                let (_name, _ext) = split_filename(&entry.file_name().to_string_lossy())?;
96
97                // Read file data for preload and calculate CRC
98                let file_data = std::fs::read(entry.path())
99                    .with_context(|| format!("Failed to read file: {}", entry.path().display()))?;
100
101                let mut hasher = Hasher::new();
102                hasher.update(&file_data);
103                let crc32 = hasher.finalize();
104
105                // For now, we embed all files (no separate archive files)
106                let metadata = FileMetadata {
107                    preload: file_data,
108                    crc32,
109                    preload_length: 0, // Will be set during save
110                    archive_index: EMBEDDED_ARCHIVE_INDEX,
111                    archive_offset: 0, // Will be set during save
112                    file_length: 0,    // Will be set during save
113                };
114
115                tree.insert(path_str, metadata);
116                // file_count += 1;
117            }
118        }
119
120        let tree_length = Self::calculate_tree_length(&tree);
121
122        let header = VPKHeader {
123            signature: VPK_SIGNATURE,
124            version: VPKVersion::V2, // Default to V2 for new files
125            tree_length,
126            header_length: 28,           // V2 header length
127            embed_chunk_length: Some(0), // Will be calculated during save
128            chunk_hashes_length: Some(0),
129            self_hashes_length: Some(48),
130            signature_length: Some(0),
131        };
132
133        Ok(VPK {
134            path: None,
135            header,
136            tree,
137            checksums: None,
138        })
139    }

141    /// Saves the VPK to the specified path
142    pub fn save<P: AsRef<Path>>(&self, output_path: P) -> Result<()> {
143        let output_path = output_path.as_ref();
144        let mut file =
145            BufWriter::new(File::create(output_path).with_context(|| {
146                format!("Failed to create VPK file: {}", output_path.display())
147            })?);
148
149        // Write header (will update embed_chunk_length later)
150        self.write_header(&mut file)?;
151        let header_end = file.stream_position()? as u32;
152
153        // Write file tree and embedded data
154        let embed_chunk_length = self.write_file_tree_and_data(&mut file)?;
155
156        // Calculate and write checksums for V2
157        if self.header.version == VPKVersion::V2 {
158            // Update embed_chunk_length in header
159            file.seek(SeekFrom::Start(12))?; // Position of embed_chunk_length
160            file.write_all(&embed_chunk_length.to_le_bytes())?;
161            file.seek(SeekFrom::End(0))?;
162
163            // Flush buffer and get underlying file for checksum calculation
164            file.flush()?;
165            let mut underlying_file = file
166                .into_inner()
167                .map_err(|e| anyhow::anyhow!("Failed to get underlying file: {}", e))?;
168
169            self.write_checksums(&mut underlying_file, header_end, embed_chunk_length)?;
170        } else {
171            file.flush()?;
172        }
173
174        Ok(())
175    }

177    /// Reads the VPK header from the file
178    fn read_header<R: Read>(reader: &mut R) -> Result<VPKHeader> {
179        let mut header_bytes = [0u8; 12];
180        reader
181            .read_exact(&mut header_bytes)
182            .context("Failed to read VPK header")?;
183
184        let signature = u32::from_le_bytes([
185            header_bytes[0],
186            header_bytes[1],
187            header_bytes[2],
188            header_bytes[3],
189        ]);
190        let version_num = u32::from_le_bytes([
191            header_bytes[4],
192            header_bytes[5],
193            header_bytes[6],
194            header_bytes[7],
195        ]);
196        let tree_length = u32::from_le_bytes([
197            header_bytes[8],
198            header_bytes[9],
199            header_bytes[10],
200            header_bytes[11],
201        ]);
202
203        if signature != VPK_SIGNATURE {
204            bail!("Invalid VPK signature: 0x{:08x}", signature);
205        }
206
207        let version = match version_num {
208            1 => VPKVersion::V1,
209            2 => VPKVersion::V2,
210            _ => bail!("Unsupported VPK version: {}", version_num),
211        };
212
213        let mut header = VPKHeader {
214            signature,
215            version,
216            tree_length,
217            header_length: 12,
218            embed_chunk_length: None,
219            chunk_hashes_length: None,
220            self_hashes_length: None,
221            signature_length: None,
222        };
223
224        // Read V2 extended header
225        if version == VPKVersion::V2 {
226            let mut v2_header = [0u8; 16];
227            reader
228                .read_exact(&mut v2_header)
229                .context("Failed to read V2 header")?;
230
231            header.embed_chunk_length = Some(u32::from_le_bytes([
232                v2_header[0],
233                v2_header[1],
234                v2_header[2],
235                v2_header[3],
236            ]));
237            header.chunk_hashes_length = Some(u32::from_le_bytes([
238                v2_header[4],
239                v2_header[5],
240                v2_header[6],
241                v2_header[7],
242            ]));
243            header.self_hashes_length = Some(u32::from_le_bytes([
244                v2_header[8],
245                v2_header[9],
246                v2_header[10],
247                v2_header[11],
248            ]));
249            header.signature_length = Some(u32::from_le_bytes([
250                v2_header[12],
251                v2_header[13],
252                v2_header[14],
253                v2_header[15],
254            ]));
255            header.header_length = 28;
256        }
257
258        Ok(header)
259    }

261    /// Reads the file tree from the VPK
262    fn read_file_tree<R: Read>(
263        reader: &mut R,
264        header: &VPKHeader,
265    ) -> Result<HashMap<String, FileMetadata>> {
266        let mut tree = HashMap::new();
267
268        loop {
269            let ext = read_cstring(reader)?;
270            if ext.is_empty() {
271                break;
272            }
273
274            loop {
275                let path = read_cstring(reader)?;
276                if path.is_empty() {
277                    break;
278                }
279
280                let normalized_path = if path == " " {
281                    String::new()
282                } else {
283                    format!("{path}/")
284                };
285
286                loop {
287                    let name = read_cstring(reader)?;
288                    if name.is_empty() {
289                        break;
290                    }
291
292                    // Read file metadata
293                    let mut metadata_bytes = [0u8; 18];
294                    reader
295                        .read_exact(&mut metadata_bytes)
296                        .context("Failed to read file metadata")?;
297
298                    let crc32 = u32::from_le_bytes([
299                        metadata_bytes[0],
300                        metadata_bytes[1],
301                        metadata_bytes[2],
302                        metadata_bytes[3],
303                    ]);
304                    let preload_length = u16::from_le_bytes([metadata_bytes[4], metadata_bytes[5]]);
305                    let archive_index = u16::from_le_bytes([metadata_bytes[6], metadata_bytes[7]]);
306                    let archive_offset = u32::from_le_bytes([
307                        metadata_bytes[8],
308                        metadata_bytes[9],
309                        metadata_bytes[10],
310                        metadata_bytes[11],
311                    ]);
312                    let file_length = u32::from_le_bytes([
313                        metadata_bytes[12],
314                        metadata_bytes[13],
315                        metadata_bytes[14],
316                        metadata_bytes[15],
317                    ]);
318                    let suffix = u16::from_le_bytes([metadata_bytes[16], metadata_bytes[17]]);
319
320                    if suffix != METADATA_SUFFIX {
321                        bail!("Invalid metadata suffix: 0x{:04x}", suffix);
322                    }
323
324                    // Adjust archive offset for embedded files
325                    let actual_archive_offset = if archive_index == EMBEDDED_ARCHIVE_INDEX {
326                        header.header_length + header.tree_length + archive_offset
327                    } else {
328                        archive_offset
329                    };
330
331                    // Read preload data
332                    let preload = if preload_length > 0 {
333                        read_exact_vec(reader, preload_length as usize)?
334                    } else {
335                        Vec::new()
336                    };
337
338                    let metadata = FileMetadata {
339                        preload,
340                        crc32,
341                        preload_length,
342                        archive_index,
343                        archive_offset: actual_archive_offset,
344                        file_length,
345                    };
346
347                    let full_path = format!("{normalized_path}{name}.{ext}");
348                    tree.insert(full_path, metadata);
349                }
350            }
351        }
352
353        Ok(tree)
354    }

356    /// Reads checksums from V2 VPK files
357    fn read_checksums<R: Read + Seek>(reader: &mut R, header: &VPKHeader) -> Result<VPKChecksums> {
358        if header.version != VPKVersion::V2 {
359            bail!("Checksums only available in VPK V2");
360        }
361
362        let embed_chunk_length = header.embed_chunk_length.unwrap_or(0);
363        let chunk_hashes_length = header.chunk_hashes_length.unwrap_or(0);
364
365        // Seek to checksums section
366        let checksums_offset =
367            header.header_length + header.tree_length + embed_chunk_length + chunk_hashes_length;
368        reader.seek(SeekFrom::Start(checksums_offset as u64))?;
369
370        let mut tree_checksum = [0u8; 16];
371        let mut chunk_hashes_checksum = [0u8; 16];
372        let mut file_checksum = [0u8; 16];
373
374        reader.read_exact(&mut tree_checksum)?;
375        reader.read_exact(&mut chunk_hashes_checksum)?;
376        reader.read_exact(&mut file_checksum)?;
377
378        Ok(VPKChecksums {
379            tree_checksum,
380            chunk_hashes_checksum,
381            file_checksum,
382        })
383    }

385    /// Writes the VPK header
386    fn write_header<W: Write>(&self, writer: &mut W) -> Result<()> {
387        writer.write_all(&self.header.signature.to_le_bytes())?;
388        writer.write_all(&(self.header.version as u32).to_le_bytes())?;
389        writer.write_all(&self.header.tree_length.to_le_bytes())?;
390
391        if self.header.version == VPKVersion::V2 {
392            writer.write_all(&self.header.embed_chunk_length.unwrap_or(0).to_le_bytes())?;
393            writer.write_all(&self.header.chunk_hashes_length.unwrap_or(0).to_le_bytes())?;
394            writer.write_all(&self.header.self_hashes_length.unwrap_or(48).to_le_bytes())?;
395            writer.write_all(&self.header.signature_length.unwrap_or(0).to_le_bytes())?;
396        }
397
398        Ok(())
399    }

401    /// Writes the file tree and embedded data
402    fn write_file_tree_and_data<W: Write + Seek>(&self, writer: &mut W) -> Result<u32> {
403        // Group files by extension and path
404        let mut grouped_files: FileHashMap = HashMap::new();
405
406        for (full_path, metadata) in &self.tree {
407            let (name, ext) = split_filename(full_path)?;
408            let path_part = if let Some(slash_pos) = name.rfind('/') {
409                name[..slash_pos].to_string()
410            } else {
411                " ".to_string() // Root directory
412            };
413            let name_part = if let Some(slash_pos) = name.rfind('/') {
414                name[slash_pos + 1..].to_string()
415            } else {
416                name
417            };
418
419            grouped_files
420                .entry(ext)
421                .or_default()
422                .entry(path_part)
423                .or_default()
424                .push((name_part, metadata));
425        }
426
427        let data_start_offset = writer.stream_position()? as u32 + self.header.tree_length;
428        let mut current_data_offset = data_start_offset;
429        let mut embed_chunk_length = 0;
430
431        // Write file tree
432        for (ext, paths) in &grouped_files {
433            write_cstring(writer, ext)?;
434
435            for (path, files) in paths {
436                write_cstring(writer, path)?;
437
438                for (name, metadata) in files {
439                    write_cstring(writer, name)?;
440
441                    // Write metadata (as above)
442                    writer.write_all(&metadata.crc32.to_le_bytes())?;
443                    writer.write_all(&0u16.to_le_bytes())?; // preload_length = 0  
444                    writer.write_all(&EMBEDDED_ARCHIVE_INDEX.to_le_bytes())?;
445                    writer.write_all(&(current_data_offset - data_start_offset).to_le_bytes())?;
446                    writer.write_all(&(metadata.preload.len() as u32).to_le_bytes())?;
447                    writer.write_all(&METADATA_SUFFIX.to_le_bytes())?;
448
449                    current_data_offset += metadata.preload.len() as u32;
450                    embed_chunk_length += metadata.preload.len() as u32;
451                }
452                writer.write_all(&[0])?; // End of files in this path
453            }
454            writer.write_all(&[0])?; // End of paths in this extension  
455        }
456        writer.write_all(&[0])?; // End of tree
457
458        // Now write all the actual file data
459        for paths in grouped_files.values() {
460            for files in paths.values() {
461                for (_name, metadata) in files {
462                    if !metadata.preload.is_empty() {
463                        writer.write_all(&metadata.preload)?;
464                    }
465                }
466            }
467        }
468
469        Ok(embed_chunk_length)
470    }

491    /// Calculates the tree length for the given file set
492    fn calculate_tree_length(tree: &HashMap<String, FileMetadata>) -> u32 {
493        let mut length = 1; // Final null terminator
494
495        // Group by extension for calculation
496        let mut extensions: HashMap<String, HashMap<String, Vec<String>>> = HashMap::new();
497
498        for full_path in tree.keys() {
499            if let Ok((name, ext)) = split_filename(full_path) {
500                let path_part = if let Some(slash_pos) = name.rfind('/') {
501                    name[..slash_pos].to_string()
502                } else {
503                    " ".to_string()
504                };
505                let name_part = if let Some(slash_pos) = name.rfind('/') {
506                    name[slash_pos + 1..].to_string()
507                } else {
508                    name
509                };
510
511                extensions
512                    .entry(ext)
513                    .or_default()
514                    .entry(path_part)
515                    .or_default()
516                    .push(name_part);
517            }
518        }
519
520        for (ext, paths) in extensions {
521            length += cstring_length(&ext) as u32;
522
523            for (path, names) in paths {
524                length += cstring_length(&path) as u32;
525
526                for name in names {
527                    length += cstring_length(&name) as u32;
528                    length += 18; // Metadata size only, no preload data in tree
529                }
530                length += 1; // Path terminator
531            }
532            length += 1; // Extension terminator
533        }
534
535        length
536    }
